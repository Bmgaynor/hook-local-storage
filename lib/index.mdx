# use-local-storage

Module for interacting with browsers localstorage

## Install

```bash
npm install hook-local-storage
```


## Usage

import { Playground, PropsTable } from 'docz'
import useLocalStorage from './use-local-storage'
import TestBed from './TestBed'


<Playground>
  {() => {
    const [count, setCount] = useLocalStorage('count', 0)
    const [todos, setTodos] = useLocalStorage('todos', [])
    const [input, setInput] = useLocalStorage('input', '')
    const addTodo = () => {
        setTodos(todos.concat(input))
        setInput('')
        
    }
    const removeTodo = (index) => {
        const copy = [].concat(todos)
        copy.splice(index, 1)
        setTodos(copy)
    }

    return (
        <TestBed>
            <h1>useLocalStorage</h1>
            <div>
                <p>Refeashing the page will keep the count!</p>
                <button nature='primary' onClick={() => setCount(count + 1)}>count: {count}</button>
            </div>
            <div> 
                <p>Works for objects too!</p>
                <input type='text' value={input} onChange={e => setInput(e.target.value)} onKeyPress={e => e.key === 'Enter' &&  addTodo()}/>
                <button nature='primary' onClick={addTodo}>addTodo</button>
                <script>alert('hi')</script>
                {todos.map((todo, index) => {
                    return <p key={index} onClick={() => removeTodo(index)}>{todo}</p>
                })}
            </div>
        </TestBed>
    )
  }}
</Playground>
